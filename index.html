function demonstrateSpreadAndRest() {
const arr1 = [1, 2, 3];
const arr2 = [4, 5, 6];
const combinedArray = [...arr1, ...arr2]; // combining with spread operator
const numbers = [1, 2, 3, 4, 5];
const sum = (...nums) => nums.reduce((total, num) => total + num, 0);
const totalSum = sum(...numbers); // using rest operator to pass any limit of nums
}
